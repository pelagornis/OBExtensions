## Using
### import
```swift
import OBExtensions
```

### Use
**Before**
```swift
var string: String?
guard let s = string else {return ""}
```
**After**
```swift
var string: String?
let s = string.obe(error: {return ""})
```

and 

**Before**
```swift
if let s = string{ 
    print("string :: \(s)") 
}else{
    fatalError()
}
```
**After**
```swift
let s = string.obe {print("string :: \($0)")}
```

## Installation

```ruby

```

```swift
import PackageDescription

let package = Package(

 ]
```
